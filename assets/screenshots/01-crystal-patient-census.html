🔍 Embedded SQL Query
SELECT w.ward_name, w.ward_code, p.patient_id, p.patient_name,
       p.admission_date, p.attending_physician, p.diagnosis_code
FROM wards w
INNER JOIN patients p ON w.ward_id = p.current_ward_id
WHERE p.discharge_date IS NULL
ORDER BY w.ward_name, p.admission_date
