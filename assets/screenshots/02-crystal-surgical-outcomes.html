🔍 Embedded SQL Query
SELECT s.surgery_id, s.surgery_date, s.procedure_code, s.surgeon_name,
       s.duration_minutes, s.outcome_code, c.complication_description
FROM surgeries s
LEFT JOIN complications c ON s.surgery_id = c.surgery_id
WHERE s.surgery_date BETWEEN :StartDate AND :EndDate
📐 Crystal Formulas
// Complication Rate — across all surgeries in range
@ComplicationRate = Count({complications.complication_id}) / Count({surgeries.surgery_id}) * 100
