🔍 Inline SQL Query
SELECT metric_name, actual_value, target_value,
       CASE
           WHEN actual_value >= target_value            THEN 'Green'
           WHEN actual_value >= target_value * 0.9  THEN 'Yellow'
           ELSE 'Red'
       END AS status_color
FROM dbo.cms_quality_metrics
WHERE reporting_period = @Period
🎨 Conditional Formatting (SSRS Expression)
// Background colour expression on MetricName textbox
=IIF(Fields!status_color.Value = "Green",  "LightGreen",
  IIF(Fields!status_color.Value = "Yellow", "Yellow",
                                              "LightCoral"))
